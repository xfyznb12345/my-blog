# 基于 TypeScript 的 Axios 接口封装

前言

本文主要介绍项目引入 typescript 下，针对 axios 的接口请求封装，统一的请求拦截，响应拦截。

# 思路

  1. 解决多个域名、多环境下的 baseUrl 混乱不好维护。
  2. 解决接入多个系统，错误码不统一，无法统一处理。
  3. 解决发起重复请求，造成服务器资源浪费。

# Axios 核心编写

  • 在 src/utils/http 下新建一个 core.ts 文件
/*
 * @Description: axios二次封装
 * @Author: XFY
 */
import axios, { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios';
import type { BaseResponse, UpmResponse, DownloadOptions, RequestOptions, UploadOptions } from './types.d';
import { axiosCanceler } from './index';
import dataTransform from './axiosDispose';
import { downloadByData } from '../download';
import { getToken } from '../cache/authCache';
import { TOKEN_KEY } from './../cache/cacheEnum';
import { message as Message } from 'ant-design-vue';
/**
 * @description 对请求参数进行处理
 * @returns config
 */
function transformConfig(config: AxiosRequestConfig, params: any) {
  const { method } = config;
  const props = ['delete', 'get'].includes(method!.toLocaleLowerCase()) ? 'params' : 'data';
  return {
    ...config,
    [props]: params,
  };
}
class EnclosureHttp {
  // 当前Axios实例对象
  private axiosInstance: AxiosInstance;
  constructor(option: AxiosRequestConfig) {
    this.axiosInstance = axios.create(option);
    this.httpInterceptors();
  }
  /**
   * @description 请求/响应拦截器
   * @return void
   */
  private httpInterceptors() {
    const { responseCatch } = dataTransform;
    this.axiosInstance.interceptors.request.use(
      (config: AxiosRequestConfig) => {
        const {
          headers: { ignoreCancelToken, cancelNewRepeatRequest },
        } = config;
        if (!ignoreCancelToken) {
          cancelNewRepeatRequest ? axiosCanceler.addFirstRequest(config) : axiosCanceler.addFinalRequest(config);
        }
        const token = getToken();
        if (token) {
          // jwt token
          config.headers[TOKEN_KEY] = token;
        }
        return config;
      },
      error => {
        return Promise.reject(error);
      }
    );
    this.axiosInstance.interceptors.response.use((response: AxiosResponse) => {
      response && axiosCanceler.removeRequest(response.config);
      return response;
    }, responseCatch);
  }
  /**
   * @description 基本业务请求
   */
  public request<T = any>(config: RequestOptions): Promise<BaseResponse<T>> {
    const { method, url, params, axiosConfig } = config;
    const conf = transformConfig({ method, url, ...axiosConfig }, params);
    return new Promise((resolve, reject) => {
      this.axiosInstance
        .request<BaseResponse<T>>(conf)
        .then(response => {
          switch (response.data.code) {
            case 20001:
              Message.error(response.data.message);
              reject(response.data);
          }
          resolve(response.data);
        })
        .catch(error => {
          reject(error);
        });
    });
  }
  /**
   * @description 文件上传
   */
  public uploadFile<T = any>(config: UploadOptions) {
    const { url, params, onUploadProgress } = config;
    const formData = new FormData();
    Object.keys(params).forEach(key => {
      formData.append(key, params[key]);
    });
    return this.request<T>({
      url,
      method: 'post',
      params: formData,
      axiosConfig: {
        headers: {
          'Content-type': 'multipart/form-data;charset=UTF-8',
          ignoreCancelToken: true,
        },
        onUploadProgress: onUploadProgress,
      },
    });
  }
  /**
   * @description 文件下载
   * @see 文件名需要由后端加注解 Access-Control-Expose-Headers : 'content-disposition'
   */
  public downLoadFile(config: DownloadOptions): Promise<Blob> {
    const { url, params } = config;
    const conf = transformConfig({ method: 'get', url, responseType: 'blob' }, params);
    return new Promise((resolve, reject) => {
      this.axiosInstance
        .request<Blob>(conf)
        .then(response => {
          const fileName = response.headers?.['content-disposition']
            ? response.headers?.['content-disposition'].split('=')[1]
            : '';
          fileName && downloadByData(response.data, decodeURIComponent(fileName));
          resolve(response.data);
        })
        .catch(error => {
          reject(error);
        });
    });
  }
}
export default EnclosureHttp;
